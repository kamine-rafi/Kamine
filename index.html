const chat = document.getElementById("chat"); const input = document.getElementById("msg"); function addMessage(text, type) { const div = document.createElement("div"); div.className = "message " + type; div.innerHTML = text; chat.appendChild(div); chat.scrollTop = chat.scrollHeight; } function send() { const text = input.value.trim(); if (text === "") return; addMessage(text, "user"); input.value = ""; setTimeout(() => { addMessage("🤖 Kamine: আমি এখনো ডেভেলপমেন্টে আছি। খুব শিগগিরই OpenAI-এর সাথে সংযুক্ত হব।", "ai"); }, 600); } input.addEventListener("keypress", function(e){ if(e.key==="Enter"){ send(); } });
